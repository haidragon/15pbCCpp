关于格式控制符：
  1. #include <stdio.h> 
  2. int main()
  3. {
  4. 	int Num = 100;
  5. 	Num = 1 + 5;
  6. 	int Num2 = 0;
  7. 	Num2 = 10 * 5;
  8. 	//      格式字符串          匹配的参数列表
  9. 	printf("运算的结果是：%d %d \n",   Num,10);
  11. 	//把格式字符串原样输出，遇到了格式控制符，去参数列表中匹配。
  12. 	//请注意：格式符的个数，要与后面的参数格式一致。否则结果不确定。
  13. 	//还可能遇到转义字符，比如\n。就会显示相应的效果。
  14. 	printf("另一个运算的结果是：%d\n", Num2);
  15. 	//我们目前，需要记忆的格式控制符：
  16. 	//%d   整数
  17. 	//%c   字符
  18. 	//%lf  双精度浮点数
  19. 	//%s   输出字符
  20. 	printf("今天天气:%s\n", "很好");
  22. 	printf("今天天气:%s\n", "不太好");
  23. 	printf("今天天气:%s\n", "下冰雹了");
  25. 	float fNum = 100.5;
  26. 	printf("%lf\n", fNum);
  27. 	printf("%d\n", fNum);//浮点数和整数，在输出的时候，格式符不要混用。
  28. 	int Num3 = 0x12345678;
  29. 	printf("%lf\n", (double)Num3);//浮点数和整数，在输出的时候，格式符不要混用。
  31. 	char cNum = 'A';
  32. 	printf("%d\n", cNum);
  33. 	printf("%c\n", 65);//这个实验说明了，字符本质上就是数字
  35. }
关于scanf_s的用法
  1. #include <stdio.h> 
  2. int main()
  3. {
  4. 	int a = 0;
  5. 	int b = 0;
  7. 	//&叫做取地址符号，它的作用是得到变量的地址
  8. 	printf("请输入两个数：");
  9. 	scanf_s("%d %d", &a,&b);//除了格式控制符之外，不要再写其他的了
  10. 	                        //尽量一个scanf中输入的数据只有一个。
  11. 	printf("%d", a);
  12. 	printf("%d", &a);       //变量名前加取地址，得到的是变量的地址，可以打印出来，看到
  13. }
