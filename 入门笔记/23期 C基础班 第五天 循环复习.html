循环结构一共有三种：
while
  1. 	while (/*判断语句*/)
  2. 	{
  3. 		/*循环体语句*/
  4. 	}
do-while
  1. 	do
  2. 	{
  3. 		/*循环体语句*/
  4. 	} while (/*判断语句*/);
for
  1. 	for (/*循环变量的初始化*/ ;/*判断语句*/; /*循环变量的变化方式*/)
  2. 	{
  3. 		/*循环体语句*/
  4. 	}
